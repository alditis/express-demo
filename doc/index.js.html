/**
 * Init models.
 * @module model/index
 */

const fs = require('fs');
const path = require('path');
const Sequelize = require('sequelize');
const db = {};

const sequelize = new Sequelize(
    PER.config.db.database,
    PER.config.db.username,
    PER.config.db.password, {
        host: PER.config.db.host,
        dialect: PER.config.db.dialect,
        pool: {
            max: PER.config.db.pool.max,
            min: PER.config.db.pool.min,
            idle: PER.config.db.pool.idle
        },
        define: {
            timestamps: true,
            paranoid: true,
            underscored: true,
            freezeTableName: true,
            createdAt: 'created',
            updatedAt: 'updated',
            deletedAt: 'deleted',
            version: 'locking'
        },
        logging: PER.helper.getDBLogging(PER.config.db.logging)
    }
);

fs.readdirSync(__dirname)
    .filter(file => {
        return !PER.helper.isHiddenPath(file) && file !== "index.js";
    })
    .forEach(file => {
        const model = sequelize.import(path.join(__dirname, file));
        db[model.name] = model;
    });

Object.keys(db).forEach(modelName => {
    if (db[modelName].associate) {
        db[modelName].associate(db);
    }
});

db.sequelize = sequelize;
db.Sequelize = Sequelize;

module.exports = db;
